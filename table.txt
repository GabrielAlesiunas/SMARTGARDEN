CREATE TABLE usuario (
    id BIGINT PRIMARY KEY 	AUTO_INCREMENT,
    nome VARCHAR(100),
    email VARCHAR(100),
    senha VARCHAR(20),
    foto_perfil VARCHAR(100)
)ENGINE=INNODB;

CREATE TABLE produto (
    id BIGINT PRIMARY KEY	AUTO_INCREMENT,
    nome VARCHAR(100),
    descricao VARCHAR(200),
    valor BIGINT
)ENGINE=INNODB;

CREATE TABLE situacaoplaca (
    id BIGINT PRIMARY KEY	AUTO_INCREMENT,
    data DATE,
    temperatura BIGINT,
    umidade BIGINT,
    luminosidade BIGINT,
    fk_placa_id BIGINT,
    FOREIGN KEY (fk_placa_id) 	REFERENCES placa(id)
)ENGINE=INNODB;

CREATE TABLE plantas (
    id BIGINT PRIMARY KEY	AUTO_INCREMENT,
    nome VARCHAR(100),
    tempideal BIGINT,
    umideal BIGINT,
    lumideal BIGINT
)ENGINE=INNODB;

CREATE TABLE placa (
    id BIGINT PRIMARY KEY	AUTO_INCREMENT,
    mac VARCHAR(200),
    descricao VARCHAR(200),
    fk_usuario_id BIGINT,
    FOREIGN KEY (fk_usuario_id) 	REFERENCES usuario(id)
)ENGINE=INNODB;
